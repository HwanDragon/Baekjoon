#include <iostream>

/*
1을 만드는 방법 : 1

2를 만드는 방법
1 + 1
2

3을 만드는 방법
1 + 1 + 1
1 + 2
2 + 1
3

4를 만드는 방법 (총 7가지)
(앞에 1이 추가되고 뒤쪽은 3을 만드는 방법의 수)
1+1+1+1
1+1+2
1+2+1
1+3

(앞에 2가 추가되고 뒤쪽은 2를 만드는 방법의 수)
2+1+1
2+2

(앞에 3이 추가되고 뒤쪽은 1을 만드는 방법의 수)
3+1
*/

// 즉, 점화식은 f(n) = f(n-1) + f(n-2) + f(n-3)
using namespace std;

int dp[12];	// dp로 미리 계산

int main()
{
	dp[1] = 1;	// 1을 만드는 방법의 수 
	dp[2] = 2;	// 2를 만드는 방법의 수
	dp[3] = 4;	// 3을 만드는 방법의 수
	for (int i = 4; i <= 11; i++)
		dp[i] = dp[i - 1] + dp[i - 2] + dp[i - 3];

	int t;
	cin >> t;
	while (t--)
	{
		int n;
		cin >> n;

		cout << dp[n] << "\n";
	}
	return 0;
}
